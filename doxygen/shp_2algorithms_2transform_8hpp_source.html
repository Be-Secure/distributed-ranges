24template <class ExecutionPolicy>
25auto transform(ExecutionPolicy &&policy, lib::distributed_range auto &&in,
26 lib::distributed_iterator auto out, auto &&fn) {
27
28 static_assert( // currently only one policy supported
29 std::is_same_v<std::remove_cvref_t<ExecutionPolicy>, device_policy>);
30
31 std::vector<sycl::event> events;
32 using OutT = typename decltype(out)::value_type;
33 std::vector<void *> buffers;
34 const auto out_end = out + rng::size(in);
35
36 for (auto &&[in_seg, out_seg] :
37 views::zip(in, rng::subrange(out, out_end)).zipped_segments()) {
38 auto in_device = policy.get_devices()[in_seg.rank()];
39 sycl::queue q(shp::context(), in_device);
40 const std::size_t seg_size = rng::size(in_seg);
41 assert(seg_size == rng::size(out_seg));
42 auto local_in_seg = __detail::get_local_segment(in_seg);
43
44 if (in_seg.rank() == out_seg.rank()) {
45 auto local_out_seg = __detail::get_local_segment(out_seg);
46 events.emplace_back(q.parallel_for(seg_size, [=](auto idx) {
47 local_out_seg[idx] = fn(local_in_seg[idx]);
48 }));
49 } else {
50 OutT *buffer =
51 sycl::malloc_device<OutT>(seg_size, in_device, shp::context());
52 buffers.push_back(buffer);
53
54 sycl::event compute_event = q.parallel_for(
55 seg_size, [=](auto idx) { buffer[idx] = fn(local_in_seg[idx]); });
56 events.emplace_back(q.copy(buffer, __detail::get_local(out_seg.begin()),
57 seg_size, compute_event));
58 }
59 }
60 __detail::wait(events);
61
62 for (auto *b : buffers)
63 sycl::free(b, shp::context());
64
65 return rng::unary_transform_result<decltype(rng::end(in)), decltype(out_end)>{
66 rng::end(in), out_end};
67}
